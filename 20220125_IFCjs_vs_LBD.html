Use it in an app


import * as WebIFC from "web-ifc/web-ifc-api.js";
import { LBDParser } from "ifc-lbd";

import * as path from 'path';
import { readFile } from "fs";
import * as util from "util";
const readFileAsync = util.promisify(readFile);

async function main(){

	// Read file
	const modelPath = path.join(__dirname, './Duplex.ifc');
	const modelData = readFileAsync(modelPath);
	
	// Init API and load model
	const ifcApi = new WebIFC.IfcAPI();
	await ifcApi.Init();
	const modelID = ifcApi.OpenModel(ifcModelData);
	
	// Init LBD Parser and parse BOT
	const lbdParser = new LBDParser();
	const botTriples = await lbdParser.parseBOTTriples(ifcApi, modelID);
	console.log(botTriples);

	// Close the model, all memory is freed
	ifcApi.CloseModel(modelID);

}

Post-processing with N3 + Comunica

FSO uses some pattern matches that require post-processing with SPARQL update queries

  1. Load data into N3 store
  2. Use Comunica to query the store
  3. Serialize store content to file with N3 stream writer
async function componentConections(): Promise<void>{
	const query = `
		PREFIX fso: 	<https://w3id.org/fso#>
		INSERT{
			?e1 fso:connectedWith ?e2 .
			?e2 fso:connectedWith ?e1 .
			?e1 fso:feedsFluidTo ?e2 .
			?e2 fso:hasFluidFedBy ?e1
		}
		WHERE{
			?e1 fso:connectedPort ?p1 .
			?p1 fso:connectedPort ?p2 .
			?p2 fso:connectedComponent ?e2 .
			?p1 a fso:OutPort .
			?p2 a fso:InPort .
		}`;
	await this.executeUpdateQuery(query);
}

Some queries use Extension Functions

async function segmentLengths(): Promise<void>{
	const query = `INSERT{
		?seg ex:length ?d
	}
	WHERE{
		?seg a fso:Segment ;
			fso:connectedPort ?port1 , ?port2 .
		FILTER(?port1 != ?port2)
		?port1 omg:hasGeometry/fog::asSfa_v2-wkt ?p1 .
		?port2 omg:hasGeometry/fog::asSfa_v2-wkt ?p2 .
		BIND(geosf:distance(?p1, ?p2, 3) AS ?d)
	}`;
	await this.executeUpdateQuery(query);
}

Some queries use Extension Functions

// GEOJSON
// geosf:distance(p1, p2, decimals)
'http://www.opengis.net/def/function/geosparql/distance'(args: RDF.Term[]) {

	// Number of decimals? (default = 8)
	const decimals = args[2] != undefined ? parseFloat(args[2].value) : 8;

	// Make sure both args are literal values
	if (args[0].termType === 'Literal' && args[1].termType === 'Literal') {

		const p1 = parseWKT(args[0].value);
		const p2 = parseWKT(args[1].value);
		
		const a = p1[0] - p2[0];
		const b = p1[1] - p2[1];
		const c = p1[2] - p2[2];

		const d: number = round(Math.sqrt(a * a + b * b + c * c), decimals);

		return DF.literal(d.toString(), DF.namedNode('http://www.w3.org/2001/XMLSchema#decimal'));

	}

	// Just return literal "ERROR" if something fails
	return DF.literal("ERROR");
}
